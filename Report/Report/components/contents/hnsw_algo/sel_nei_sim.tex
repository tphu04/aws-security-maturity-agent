\begin{algorithm}[H]
\caption{SELECT-NEIGHBORS-SIMPLE($q$, $C$, $M$)}
\begin{algorithmic}[1]
\Require Base element $q$, candidate elements $C$, number of neighbors to return $M$
\Ensure $M$ nearest elements to $q$
\State \textbf{return} $M$ nearest elements from $C$ to $q$
\end{algorithmic}
\end{algorithm}
